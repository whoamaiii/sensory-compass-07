            return isNaN(dt.getTime()) ? new Date() : dt;

          return new Date();

          logger.error('Error coercing timestamp:', v, error);
          return new Date();

        logger.error('Error normalizing filteredData:', error);
        return {
          entries: [],
          emotions: [],
          sensoryInputs: []
